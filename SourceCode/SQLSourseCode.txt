USE chefdishmanagement;

CREATE TABLE Chef (
    ChefID INT PRIMARY KEY,
    Name VARCHAR(50),
    Specialty VARCHAR(100),
    Password VARCHAR(50)
);
CREATE TABLE Dish (
    DishID INT PRIMARY KEY,
    Name VARCHAR(50),
    ChefID INT,
    Description VARCHAR(255),
    FOREIGN KEY (ChefID) REFERENCES Chef(ChefID)
);
CREATE TABLE Admin (
    AdminID INT PRIMARY KEY,
    Name VARCHAR(50),
    Password VARCHAR(50)
);
